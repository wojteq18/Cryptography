@startuml

class Card {
    - suit: String
    - rank: String
    + getSuit(): String
    + getRank(): String
    + getValue(): int
}

class Deck {
    - cards: List<Card>
    + shuffle(): void
    + draw(): Card
    + resetDeck(): void
}

class Player {
    - name: String
    - hand: List<Card>
    - score: int
    + getName(): String
    + addCardToHand(c: Card): void
    + addPoint(p: int): void
    + playCard(): Card
    + getScore(): int
}

class GameRound {
    - players: List<Player>
    - deck: Deck
    + startRound(): void
    + determineRoundWinner(): Player
    + distributePoints(winner: Player): void
}

// Relacje pomiędzy klasami
Deck "1" *-- "many" Card : contains
Player "1" *-- "many" Card : has
GameRound "1" o-- "many" Player : involves
GameRound "1" --> "1" Deck : uses

@enduml
